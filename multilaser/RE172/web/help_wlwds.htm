<% multilang("Configure WDS (Wireless Distribution System) only"); %>
1. <% multilang("From the head menu, click on Wlan1."); %>
2. <% multilang("From the left-hand menu, click on Basic Settings."); %>
3. <% multilang("From the Mode drop-down list, select WDS."); %>
4. <% multilang("From the Channel Number drop-down list, select a Channel."); %>
5. <% multilang("Click Save & Apply button."); %>
6. <% multilang("Change setting successfully! Please wait 20 seconds..."); %>
7. <% multilang("From the head menu, click on Wlan1."); %>
8. <% multilang("From the left-hand menu, click on WDS settings."); %>
9. <% multilang("Check on the option Enable WDS."); %>
10. <% multilang("Click the Set Security."); %>
11. <% multilang("This page allows you setup the wireless security for WDS. When enabled, you must make sure each WDS device has adopted the same encryption algorithm and Key."); %>
12. <% multilang("Configure each field with the Encryption that you selected."); %>
13. <% multilang("Click Save & Apply button."); %>
14. <% multilang("Change setting successfully! Please wait 20 seconds..."); %>
15. <% multilang("From the head menu, click on Wlan1."); %>
16. <% multilang("From the left-hand menu, click on WDS settings."); %>
17. <% multilang("Check on the option Enable WDS."); %>
18. <% multilang("Enter the MAC Address."); %>
19. <% multilang("Enter the Comment."); %>
20. <% multilang("Click the Save & Apply."); %>
21. <% multilang("Change setting successfully! Please wait 20 seconds..."); %>
22. <% multilang("From the head menu, click on Wlan1."); %>
23. <% multilang("From the left-hand menu, click on WDS settings."); %>
24. <% multilang("The MAC Address that you created has been added in the Current Access Control List."); %>
